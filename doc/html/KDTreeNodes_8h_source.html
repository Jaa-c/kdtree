79  Leaf(Inner *parent, std::vector< Point<D>* > bucket) : Node(true, parent), bucket(bucket) {
80  for(int d = 0; d < D; d++) {
81  min[d] = std::numeric_limits<float>::max();
82  max[d] = 0;
83  }
84  for(typename std::vector< Point<D> * >::iterator it = bucket.begin(); it!= bucket.end(); ++it) {
85  Point<D> * p = *it;
86  for(int d = 0; d < D; d++) {
87  if((*p)[d] > max[d]) max[d] = (*p)[d];
88  if((*p)[d] < min[d]) min[d] = (*p)[d];
89  }
90  }
91  }
92  ~Leaf() {}
93 
94  void add(Point<D> * p) {
95  bucket.push_back(p);
96  for(int d = 0; d < D; d++) {
97  if((*p)[d] > max[d]) max[d] = (*p)[d];
98  if((*p)[d] < min[d]) min[d] = (*p)[d];
99  }
100  }
